```rust
pub fn streaming_tok_per_sec(&self) -> Option<f64> {
    let start = self.streaming_start?;
    let elapsed = start.elapsed().as_secs_f64();

    let tokens = self.streaming_chars as f64 / 4.0;
    Some(tokens / elapsed)
}
```

## Enhancement 7: Fuzzy/substring slash-command matching (Impact:3, Effort:1)

### Problem
In `src/tui/app/state/slash_suggest.rs`, `refresh_slash_suggestions` uses strict `starts_with`. Typing `/age` doesn't find `/agents`. Typing `/chat` doesn't find `/autochat`.

### Fix

In `src/tui/app/state/slash_suggest.rs`, update `refresh_slash_suggestions`:

```rust
pub fn refresh_slash_suggestions(&mut self) {
    if !self.input.starts_with('/') {
        self.slash_suggestions.clear();
        self.selected_slash_suggestion = 0;
        return;
    }
    let raw_query = self.input.split_whitespace().next().unwrap_or("");
    let query = crate::tui::app::text::normalize_slash_command(raw_query).to_lowercase();
    
    // Primary: prefix match (exact prefix)
    let mut prefix_matches: Vec<&str> = super::slash_commands::SLASH_COMMANDS
        .iter()
        .filter(|cmd| cmd.starts_with(&query))
        .copied()
        .collect();
    
    // Fallback: substring match if no prefix matches (but not for bare "/")
    if prefix_matches.is_empty() && query != "/" {
        prefix_matches = super::slash_commands::SLASH_COMMANDS
            .iter()
            .filter(|cmd| cmd.contains(&query[1..])) // strip leading /
            .copied()
            .collect();
    }
    
    // Show all if just "/"
    if prefix_matches.is_empty() && query == "/" {
        prefix_matches = super::slash_commands::SLASH_COMMANDS
            .iter()
            .copied()
            .collect();
    }
    
    self.slash_suggestions = prefix_matches.into_iter().map(String::from).collect();
    if self.selected_slash_suggestion >= self.slash_suggestions.len() {
        self.selected_slash_suggestion = self.slash_suggestions.len().saturating_sub(1);
    }
}
```

Replace the ENTIRE `refresh_slash_suggestions` method. Keep the `slash_suggestions_visible` and other methods unchanged.

## Enhancement 8: @mention routing stub for spawned agents (Impact:4, Effort:3)

### Fix

```rust
//! Detect and route @mention messages to spawned sub-agents.

/// Extract an @mention target from input text.
/// Returns `Some((agent_name, message))` if the input starts with @mention.
/// Returns `None` if no @mention detected.
pub fn parse_mention(input: &str) -> Option<(String, String)> {
    let trimmed = input.trim();
    if !trimmed.starts_with('@') { return None; }
    let rest = &trimmed[1..]; // strip @
    let space_pos = rest.find(' ')?;
    let name = rest[..space_pos].to_string();
    let msg = rest[space_pos..].trim().to_string();
    if name.is_empty() || msg.is_empty() { return None; }
    Some((name, msg))
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn parses_mention() {
        let (name, msg) = parse_mention("@forge fix the tests").unwrap();
        assert_eq!(name, "forge");
        assert_eq!(msg, "fix the tests");
    }

    #[test]
    fn no_mention() {
        assert!(parse_mention("regular message").is_none());
    }

    #[test]
    fn empty_message() {
        assert!(parse_mention("@forge").is_none());
    }
}
```
