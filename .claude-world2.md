## Step 2: Create src/github_pr/provenance_block.rs

This module generates a markdown verification block for PR bodies:

```rust
//! Generate provenance verification blocks for PR bodies.

use crate::provenance::ExecutionProvenance;

/// Build a markdown verification block from provenance metadata.
pub fn provenance_markdown_block(provenance: &ExecutionProvenance) -> String {
    let sig = crate::provenance::signature::sign_provenance(provenance);
    let sig_label = sig.as_deref().unwrap_or("UNSIGNED");
    format!(
        "### 🔐 Provenance\n\
         \n\
         | Field | Value |\n\
         |-------|-------|\n\
         | Agent | `{}` |\n\
         | Origin | `{}` |\n\
         | Run ID | `{}` |\n\
         | Model | `{}` |\n\
         | Signature | `{}` |\n\
         \n\
         > Verify: `codetether verify-pr {}`",
        provenance.identity.agent_name,
        provenance.identity.origin.as_str(),
        provenance.run_id.as_deref().unwrap_or("N/A"),
        provenance.identity.model.as_deref().unwrap_or("unknown"),
        &sig_label[..sig_label.len().min(16)],
        provenance.provenance_id,
    )
}
```

Adjust field names to match the actual ExecutionProvenance struct you read. Use Option::as_deref() patterns for optional fields.

## Step 3: Create script/verify_pr.sh

A standalone CI-side verifier that checks provenance signatures in git trailers:

```bash
#!/usr/bin/env bash
# Verify codetether provenance signatures in a PR.
# Usage: ./verify_pr.sh <commit-range or PR number>
set -euo pipefail

RANGE="${1:?Usage: verify_pr.sh <commit-range>}"

echo "🔐 Verifying codetether provenance for: $RANGE"
echo ""

COMMITS=$(git log --format="%H" "$RANGE")
TOTAL=0
SIGNED=0
UNSIGNED=0

for COMMIT in $COMMITS; do
    TOTAL=$((TOTAL + 1))
    TRAILER=$(git log -1 --format="%b" "$COMMIT" | grep -i "Provenance-ID:" || true)
    if [ -n "$TRAILER" ]; then
        SIGNED=$((SIGNED + 1))
        AGENT=$(git log -1 --format="%b" "$COMMIT" | grep -i "Agent-Name:" || echo "unknown")
        echo "  ✅ $COMMIT — $AGENT"
    else
        UNSIGNED=$((UNSIGNED + 1))
        echo "  ⚠️  $COMMIT — no provenance trailer"
    fi
done

echo ""
echo "Results: $SIGNED/$TOTAL signed, $UNSIGNED unsigned"

if [ "$UNSIGNED" -gt 0 ]; then
    echo "❌ FAIL: $UNSIGNED commits missing provenance"
    exit 1
fi

echo "✅ PASS: All commits have provenance trailers"
```

## Step 6: Create .github/workflows/verify_provenance.yml

A GitHub Action for CI verification:

```yaml
name: Verify Provenance
on:
  pull_request:
    types: [opened, synchronize]
jobs:
  verify:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 0
      - name: Verify provenance trailers
        run: |
          chmod +x script/verify_pr.sh
          ./script/verify_pr.sh origin/main..HEAD
```
